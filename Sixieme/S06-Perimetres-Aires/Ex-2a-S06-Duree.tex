\item Dans 3 h, il y a \dotfill min ;
\item Dans 2 jours, il y a \dotfill h ;
\item Au mois de mars, il y a \dotfill jours ;
\item Dans un siècle, il y a \dotfill mois.
\end{enumerate}
\end{exercice}
