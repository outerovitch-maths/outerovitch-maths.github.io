\colonnesep{0}

\exercice \corrige \\
\textbf{Compléter le tableau suivant par linéarité.}
\mc{
\begin{tabular}{|c|c|c|c|}
\hline
Quantité A & 5 & 11 & 6 \\ \hline
Quantité B & 10 & 22 & \kern1cm \\ \hline
\end{tabular}

\columnbreak

\phantom{0} \\
\circled{1} On remarque que $11 - 5 = 6$  \\
\circled{2} On soustrait donc $22 - 10 = 12$
}
